题目一

替换空格

　　请实现一个函数，将一个字符串中的每个空格替换成“%20”。例如，当字符串为We Are Happy.则经过替换之后的字符串为We%20Are%20Happy。

思路讲解

　　假如我们从前向后扫描数组，遇到空格就替换的话，那么需要承担O(n^2)的时间复杂度，代价太大。既然如此，我们可以试着从后向前扫描。首先遍历一遍数组，找出有多少个空格，一个空格就表示原来空格地方长度从一变成了三，那么我们便可以知道新的字符串长度。然后我们使用两个指针，一个指针p1（假设p1）放在原来字符串长度的位置，另一个指针p2放在新的长度的位置。两个指针同时向前移动，移动的同时把p1指向的值赋值给p2，当p1指向第一个空格的时候，p2移动三格并插入’%20’，p1移动一格，然后继续这个循环，知道p1 = p2为止。

代码实现

class Solution {
public:
	void replaceSpace(char* str,int length ){
		if (str == 0 || length <= 0) {//如果数组为空或者字符串长度小于0，直接返回
			return;
		}
		int numbers = 0;
		int orilength = 0;
		int i = 0;
		while (str[i] != '\0') {//从前向后遍历知道遇到空字符为止
			orilength++;
			if (str[i] == ' ') {
				numbers++;
			}
			i++;//找到有多少个空格
		}
		int newlength = orilength + numbers * 2;//新字符串的长度
		if (newlength > length) {
			return;
		}
		while (orilength >= 0 && newlength > orilength) {//当指向原始长度的指针小于0或者两个指针相等时跳出循环
			if (str[orilength] == ' ') {//原始指针遇到空格
				str[newlength--] = '0';
				str[newlength--] = '2';
				str[newlength--] = '%';
				orilength--;
			}
			else {
				str[newlength--] = str[orilength];
				orilength--;//把原始指针指向的值赋值给新指针指向的位置
			}
		}
		return;
	}
};

题目二

从尾到头打印链表

　　输入一个链表，按链表从尾到头的顺序返回一个ArrayList。

思路讲解

　　逆序打印链表，就是先进后出，那么就是栈的结构，我们可以把链表压入栈中，之后再依次弹出，便可以从尾到头打印一个链表。

代码实现

struct ListNode//定义结构体
{
	int val;
	struct ListNode* next;
	ListNode(int x) :val(x), next(NULL) {
	}
};
class Solution {
public:
	vector<int> printListFromTailToHead(ListNode* head) {
		vector<int> value;//使用vector类定义一个动态数组value
		ListNode* p = NULL;
		p = head;
		stack<int> stk;
		while (p != NULL) {//如果链表的内容不为空
			stk.push(p->val);//把链表的值压入栈中
			p = p->next;//链表指向下一个
		}
		while (!stk.empty()) {//如果栈不为空
			value.push_back(stk.top());
			stk.pop();//栈中元素从栈顶依次弹出并被压入到value
		}
		return value;
	}
};
